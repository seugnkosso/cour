1. Optimisation des performances

1.1 Utilisation de documentFragment pour les opérations massives

Lorsque vous devez ajouter un grand nombre d'éléments au DOM, utilisez un documentFragment pour minimiser les reflows et repaints, ce qui améliore les performances.

javascript
let fragment = document.createDocumentFragment();

for (let i = 0; i < 1000; i++) {
    let nouvelElement = document.createElement('div');
    nouvelElement.textContent = "Élément " + i;
    fragment.appendChild(nouvelElement);
}

document.body.appendChild(fragment);

1.2 Limitation des opérations de style et de layout

Minimisez les changements de style et de layout, car ils peuvent déclencher des reflows coûteux en performances. Effectuez autant de modifications que possible hors du flux de rendu en utilisant des classes CSS.

javascript
// Mauvaise pratique (causant des reflows multiples)
for (let i = 0; i < 100; i++) {
    document.getElementById('element' + i).style.width = '100px';
}

// Bonne pratique (un seul reflow)
for (let i = 0; i < 100; i++) {
    document.getElementById('element' + i).classList.add('largeur-100');
}

2. Gestion des événements après ajout d'éléments

2.1 Utilisation de event delegation

Au lieu d'attacher un gestionnaire d'événements à chaque élément individuel, utilisez la délégation d'événements pour réduire le nombre de gestionnaires d'événements et améliorer les performances.

javascript
// Mauvaise pratique (ajout de gestionnaires à chaque élément)
for (let i = 0; i < 100; i++) {
    let nouvelElement = document.createElement('button');
    nouvelElement.textContent = "Bouton " + i;
    nouvelElement.addEventListener('click', () => {
        // Gestionnaire d'événements
        console.log("Bouton cliqué !");
    });
    document.body.appendChild(nouvelElement);
}

// Bonne pratique (utilisation de la délégation d'événements)
document.body.addEventListener('click', (event) => {
    if (event.target.tagName === 'BUTTON') {
        console.log("Bouton cliqué !");
    }
});

3. Manipulation sécurisée du DOM

3.1 Validation des éléments avant manipulation

Vérifiez toujours si les éléments que vous essayez de manipuler existent avant de les manipuler. Cela évite les erreurs et rend votre code plus robuste.

javascript
let monElement = document.getElementById('monElement');

if (monElement) {
    // Manipulation sécurisée
    monElement.textContent = "Nouveau texte";
} else {
    console.error("L'élément n'existe pas.");
}

3.2 Utilisation de textContent pour manipuler le texte

Préférez utiliser la propriété textContent plutôt que innerHTML lorsque vous manipulez du texte pour éviter les attaques XSS (cross-site scripting).

javascript
// Mauvaise pratique (risque XSS)
let userInput = "<script>attaquer()</script>";
document.getElementById('monElement').innerHTML = userInput;

// Bonne pratique (utilisation de textContent)
document.getElementById('monElement').textContent = userInput;

2. Mise en œuvre d'un formulaire interactif

2.1 Création d'un formulaire interactif avec validation

html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Formulaire Interactif</title>
    <style>
        .error {
            color: red;
        }
    </style>
</head>
<body>
    <h1>Formulaire Interactif</h1>
    <form id="monFormulaire" onsubmit="validerFormulaire(event)">
        <label for="nom">Nom:</label>
        <input type="text" id="nom" name="nom">
        <span id="erreurNom" class="error"></span><br>

        <label for="email">Email:</label>
        <input type="email" id="email" name="email">
        <span id="erreurEmail" class="error"></span><br>

        <input type="submit" value="Soumettre">
    </form>

    <script>
        function validerFormulaire(event) {
            event.preventDefault();

            let nom = document.getElementById('nom').value;
            let email = document.getElementById('email').value;

            // Validation du nom
            if (nom.trim() === '') {
                document.getElementById('erreurNom').textContent = 'Le nom est requis.';
            } else {
                document.getElementById('erreurNom').textContent = '';
            }

            // Validation de l'email
            const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
            if (!emailRegex.test(email)) {
                document.getElementById('erreurEmail').textContent = 'Email invalide.';
            } else {
                document.getElementById('erreurEmail').textContent = '';
            }

            // Envoyer le formulaire si tout est valide
            if (nom.trim() !== '' && emailRegex.test(email)) {
                document.getElementById('monFormulaire').submit();
            }
        }
    </script>
</body>
</html>

3. Réorganisation dynamique des éléments de la page

3.1 Réorganisation des cartes avec glisser-déposer

html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">

            display: inline-block;
            cursor: grab;
        }
    </style>
</head>
<body>
    <h1>Réorganisation Dynamique</h1>
    <div id="zoneCartes">
        <div class="carte" draggable="true" ondragstart="debutGlisser(event)">Carte 1</div>
        <div class="carte" draggable="true" ondragstart="debutGlisser(event)">Carte 2</div>
        <div class="carte" draggable="true" ondragstart="debutGlisser(event)">Carte 3</div>
    </div>

    <script>
        function debutGlisser(event) {
            event.dataTransfer.setData("text/plain", event.target.innerHTML);
        }

        function permettreDeposer(event) {
            event.preventDefault();
        }

        function deposer(event) {
            event.preventDefault();
            let donnee = event.dataTransfer.getData("text/plain");
            let nouvelElement = document.createElement('div');
            nouvelElement.className = 'carte';
            nouvelElement.innerHTML = donnee;
            nouvelElement.draggable = true;
            nouvelElement.ondragstart = function (e) {
                debutGlisser(e);
            };

            document.getElementById('zoneCartes').appendChild(nouvelElement);
        }
    </script>
</body>
</html>
