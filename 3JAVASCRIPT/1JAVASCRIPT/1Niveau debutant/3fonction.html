javascript
var monTableau = [1, 2, 3, 4];

    return element > 2;
});

La méthode reduce() permet de réduire un tableau à une seule valeur en appliquant une fonction accumulateur.

javascript
var monTableau = [1, 2, 3, 4];

Les Promises sont des objets qui représentent l'achèvement ou l'échec d'une opération asynchrone.

javascript
function exemplePromise() {
    return new Promise(function(resolve, reject) {
        // Simuler une opération asynchrone
        setTimeout(function() {
            var reussi = true; // Changez-le à false pour simuler un échec
            if (reussi) {
                resolve("L'opération a réussi !");
            } else {
                reject("L'opération a échoué.");
            }
        }, 2000);
    });
}

exemplePromise()
    .then(function(resultat) {
        console.log(resultat);
    })
    .catch(function(erreur) {
        console.error(erreur);
    });

2. async/await

async est utilisé pour définir une fonction asynchrone, et await est utilisé pour attendre qu'une Promesse soit résolue.

javascript
async function exempleAsyncAwait() {
    try {
        var resultat = await exemplePromise();
        console.log(resultat);
    } catch (erreur) {
        console.error(erreur);
    }
}

exempleAsyncAwait();

Explorons en détail les différentes méthodes de stockage client en JavaScript, notamment localStorage, sessionStorage, et les cookies.

1. localStorage

localStorage permet de stocker des données sous forme de paires clé-valeur dans le navigateur. Les données stockées dans localStorage persistent même après la fermeture du navigateur.

Stockage de données :

javascript
// Enregistrement de données dans localStorage
localStorage.setItem('nom', 'John');
localStorage.setItem('age', '25');

Récupération de données :

javascript
// Récupération de données depuis localStorage
var nom = localStorage.getItem('nom');
var age = localStorage.getItem('age');

console.log(nom); // Affiche "John"
console.log(age); // Affiche "25"

2. sessionStorage

sessionStorage est similaire à localStorage, mais les données stockées dans sessionStorage sont valables uniquement pour la durée de vie de la session du navigateur. Elles sont effacées lorsque la fenêtre ou l'onglet est fermé.

Stockage de données :

javascript
// Enregistrement de données dans sessionStorage
sessionStorage.setItem('ville', 'Paris');
sessionStorage.setItem('langue', 'français');

Récupération de données :

javascript
// Récupération de données depuis sessionStorage
var ville = sessionStorage.getItem('ville');
var langue = sessionStorage.getItem('langue');

console.log(ville); // Affiche "Paris"
console.log(langue); // Affiche "français"

3. Cookies

Les cookies sont de petites informations stockées sur le côté client, mais ils ont une taille limitée (généralement 4 KB par domaine). Les cookies peuvent être définis avec une date d'expiration.

Stockage de données :

javascript
// Enregistrement de cookies
document.cookie = "nom=John; expires=Fri, 31 Dec 2022 23:59:59 GMT";
document.cookie = "age=25; expires=Fri, 31 Dec 2022 23:59:59 GMT";

Récupération de données :

javascript
// Récupération de cookies
function getCookie(nom) {
    var cookies = document.cookie.split('; ');
    for (var i = 0; i < cookies.length; i++) {
        var cookie = cookies[i].split('=');
        if (cookie[0] === nom) {
            return cookie[1];
        }
    }
    return null;
}

var nomCookie = getCookie('nom');
var ageCookie = getCookie('age');

console.log(nomCookie); // Affiche "John"
console.log(ageCookie); // Affiche "25"

Les choix entre localStorage, sessionStorage, et les cookies dépendent des besoins spécifiques de votre application. Utilisez localStorage pour les données qui doivent persister à travers les sessions, sessionStorage pour les données temporaires, et les cookies pour des données plus petites avec une date d'expiration. Pratiquez ces concepts pour bien comprendre leur fonctionnement.

Envoi d'une requête POST avec des données :

javascript
var xhr = new XMLHttpRequest();
xhr.onreadystatechange = function() {
    if (xhr.readyState === 4 && xhr.status === 200) {
        console.log(xhr.responseText);
    }
};

xhr.open('POST', 'https://api.example.com/save', true);
xhr.setRequestHeader('Content-Type', 'application/json');
var data = JSON.stringify({ nom: 'John', age: 25 });
xhr.send(data);

2. fetch()

fetch() est une API plus moderne pour effectuer des requêtes réseau. Elle renvoie une Promesse qui résout en Response à la requête.

Envoi d'une requête GET :

javascript
fetch('https://api.example.com/data')
    .then(response => response.json())

javascript

    body: JSON.stringify({ nom: 'John', age: 25 })

    .then(response => response.json())
    .then(data => console.log(data))
    .catch(error => console.error('Erreur :', error));

fetch() utilise une syntaxe plus simple et basée sur les Promesses par rapport à XMLHttpRequest. Elle prend en charge les Promesses, ce qui rend le code plus lisible et plus facile à comprendre.

Ces deux méthodes sont souvent utilisées dans les applications web modernes pour échanger des données entre le client et le serveur, notamment lors de l'envoi et de la réception de données sous forme de chaînes JSON. Veillez à utiliser JSON.parse() pour convertir des données JSON en objets JavaScript, et JSON.stringify() pour convertir des objets JavaScript en chaînes JSON.

Les modules sont un moyen de structurer le code JavaScript en le divisant en fichiers distincts et réutilisables. L'introduction des modules en JavaScript a apporté une meilleure organisation et une gestion plus propre des dépendances. Explorons comment utiliser import et export pour travailler avec des modules.

1. export

La déclaration export est utilisée pour exporter des fonctions, variables ou objets à partir d'un module.

Dans moduleA.js :

javascript
// moduleA.js
export const nom = "John";
export function direBonjour() {
    console.log("Bonjour !");
}

Dans moduleB.js :

javascript
// moduleB.js
export const age = 25;

2. import

La déclaration import est utilisée pour importer des fonctions, variables ou objets depuis un module.

Dans script.js :

javascript
// script.js
import { nom, direBonjour } from './moduleA.js';
import { age } from './moduleB.js';

console.log(nom);        // Affiche "John"
direBonjour();           // Affiche "Bonjour !"
console.log(age);        // Affiche 25

Importation avec alias :

javascript
// script.js
import { nom as prenom } from './moduleA.js';

console.log(prenom);     // Affiche "John"

Remarques importantes :

  • Les modules sont asynchrones, ce qui signifie que l'importation peut être retardée jusqu'à ce que le module soit prêt.

  • Les modules sont chargés en mode strict par défaut.

  • Les URLs de module doivent avoir une extension (par exemple, .js) pour être correctement identifiées.

  • Les modules peuvent également être chargés de manière dynamique à l'aide de import().

javascript
// Chargement dynamique d'un module
import('./moduleC.js')
    .then(moduleC => {
        console.log(moduleC.resultat);
    })
    .catch(error => {
        console.error('Erreur de chargement :', error);
    });

L'utilisation de modules facilite la gestion du code à mesure que les applications deviennent plus complexes. Ils encouragent également la réutilisation du code et la séparation des préoccupations. Pratiquez l'utilisation de modules pour une meilleure organisation de votre code JavaScript.
